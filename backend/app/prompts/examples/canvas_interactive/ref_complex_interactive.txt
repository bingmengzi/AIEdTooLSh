    <script>
        let currentTheme = 'pink';
        let isAnimating = false;
        let animationId = null;
        let recordCount = 0;

        const state = { f: 10.0, u: 30.0, pxPerCm: 15, isDragging: false, candleH: 95 };
        const themeConfig = {
            pink: { candleStripe: '#FDA4AF', candleStroke: '#FECDD3', lensFill: 'rgba(167, 243, 208, 0.2)', lensStroke: '#34D399', rayVirtual: '#FDA4AF' },
            blue: { candleStripe: '#60A5FA', candleStroke: '#BFDBFE', lensFill: 'rgba(232, 121, 249, 0.15)', lensStroke: '#C084FC', rayVirtual: '#93C5FD' }
        };

        const canvas = document.getElementById('stage');
        const ctx = canvas.getContext('2d');

        function init() {
            window.addEventListener('resize', resize);
            resize();
            setupInputs();
            canvas.addEventListener('mousedown', startDrag);
            canvas.addEventListener('touchstart', startDrag, {passive: false});
            window.addEventListener('mousemove', onDrag);
            window.addEventListener('touchmove', onDrag, {passive: false});
            window.addEventListener('mouseup', endDrag);
            window.addEventListener('touchend', endDrag);
            document.getElementById('anim-speed').addEventListener('input', e => { document.getElementById('speed-display').innerText = e.target.value + 'x'; });
            switchTab('exp');
            renderLoop(); 
        }

        function renderLoop() {
            if (!isAnimating && !state.isDragging) draw();
            requestAnimationFrame(renderLoop);
        }

        function setupInputs() {
            const updateF = (v) => { if(isNaN(v))return; state.f = Math.max(5, Math.min(15, parseFloat(v))); updateAllUI(); };
            document.querySelectorAll('.inp-range-f, .inp-num-f').forEach(el => el.addEventListener('input', e => updateF(e.target.value)));
            
            const updateU = (v) => { if(isNaN(v))return; stopAnimation(); state.u = Math.max(2, Math.min(60, parseFloat(v))); updateAllUI(); };
            document.querySelectorAll('.inp-range-u, .inp-num-u').forEach(el => el.addEventListener('input', e => updateU(e.target.value)));

            const updateV = (v) => {
                if(isNaN(v)) return; let val = parseFloat(v);
                if(Math.abs(val - state.f) < 0.5) val = state.f + 0.5;
                let newU = (val * state.f) / (val - state.f);
                if(newU < 2) newU = 2; if(newU > 60) newU = 60;
                stopAnimation(); state.u = newU; updateAllUI();
            };
            document.querySelectorAll('.inp-range-v').forEach(el => el.addEventListener('input', e => updateV(e.target.value)));
            document.querySelectorAll('.inp-num-v').forEach(el => el.addEventListener('change', e => updateV(e.target.value))); 
        }

        function switchTab(id) {
            ['exp', 'know', 'anim'].forEach(tab => {
                const btn = document.getElementById(`btn-${tab}`);
                const panel = document.getElementById(`panel-${tab}`);
                if (tab === id) {
                    btn.style.backgroundColor = currentTheme === 'pink' ? '#FBCFE8' : '#DBEAFE';
                    btn.style.color = currentTheme === 'pink' ? '#BE185D' : '#1D4ED8';
                    panel.classList.add('active');
                } else {
                    btn.style.backgroundColor = 'transparent';
                    btn.style.color = 'var(--tab-inactive)';
                    panel.classList.remove('active');
                }
            });
            if (id !== 'anim' && isAnimating) toggleAnimation();
        }

        function calc() {
            const { f, u } = state;
            let v = Infinity, m = Infinity, type = 'parallel', app = '--';
            if (Math.abs(u - f) < 0.1) {
                type = 'parallel'; app = '探照灯';
            } else {
                v = (u * f) / (u - f); m = -v / u; type = v > 0 ? 'real' : 'virtual';
                if (Math.abs(u - 2*f) < 0.1) app = '测焦距';
                else if (u > 2*f) app = '照相机';
                else if (u > f && u < 2*f) app = '投影仪';
                else if (u < f) app = '放大镜';
            }
            return { v, m, type, app };
        }

        function updateAllUI() {
            const res = calc();
            document.querySelectorAll('.val-f').forEach(el => el.textContent = state.f.toFixed(1));
            document.querySelectorAll('.val-u').forEach(el => el.textContent = state.u.toFixed(1));
            const active = document.activeElement;
            if(!active.classList.contains('inp-range-f') && !active.classList.contains('inp-num-f')) {
                document.querySelectorAll('.inp-range-f').forEach(el => el.value = state.f);
                document.querySelectorAll('.inp-num-f').forEach(el => el.value = state.f.toFixed(1));
            }
            if(!active.classList.contains('inp-range-u') && !active.classList.contains('inp-num-u')) {
                document.querySelectorAll('.inp-range-u').forEach(el => el.value = state.u);
                document.querySelectorAll('.inp-num-u').forEach(el => el.value = state.u.toFixed(1));
            }
            if(!active.classList.contains('inp-range-v') && !active.classList.contains('inp-num-v')) {
                const vDisplay = (res.v === Infinity || res.v < 0) ? 100 : res.v;
                document.querySelectorAll('.inp-range-v').forEach(el => el.value = vDisplay);
                document.querySelectorAll('.inp-num-v').forEach(el => el.value = vDisplay.toFixed(1));
            }
        }

        function toggleAnimation() { if (isAnimating) stopAnimation(); else startAnimation(); }
        function startAnimation() {
            if (isAnimating) return;
            isAnimating = true;
            document.getElementById('btn-play').innerHTML = '<i class="fa-solid fa-pause"></i>';
            document.getElementById('btn-play').classList.add('animate-pulse');
            if(state.u < 3) state.u = 60;
            let lastTime = performance.now();
            function loop(time) {
                if (!isAnimating) return;
                const dt = time - lastTime; lastTime = time;
                const speedMult = Number(document.getElementById('anim-speed').value);
                state.u -= 0.02 * dt * speedMult;
                if (state.u < 2.5) state.u = 60.0;
                updateAllUI(); draw();
                animationId = requestAnimationFrame(loop);
            }
            animationId = requestAnimationFrame(loop);
        }
        function stopAnimation() {
            isAnimating = false;
            cancelAnimationFrame(animationId);
            document.getElementById('btn-play').innerHTML = '<i class="fa-solid fa-play ml-1"></i>';
            document.getElementById('btn-play').classList.remove('animate-pulse');
        }

        function record() {
            recordCount++;
            const res = calc();
            let nature = "";
            if(res.type === 'parallel') nature = "不成像 (平行光)";
            else nature = (res.type === 'real' ? '倒立 ' : '正立 ') + (Math.abs(res.m) > 1.05 ? '放大 ' : (Math.abs(res.m) < 0.95 ? '缩小 ' : '等大 ')) + (res.type === 'real' ? '实像' : '虚像');
            const rowId = 'row-' + Date.now();
            const rowHtml = `<td class="p-2 pl-3 text-gray-400 font-mono">${recordCount}</td><td class="p-2 font-bold font-mono" style="color: var(--primary);">${state.u.toFixed(1)}</td><td class="p-2 text-gray-500 font-mono">${res.v===Infinity?'∞':Math.abs(res.v).toFixed(1)}</td><td class="p-2 text-gray-500 text-[10px] whitespace-nowrap">${nature}</td><td class="p-2 text-gray-500 text-[10px] whitespace-nowrap">${res.app}</td><td class="p-2 text-center"><button onclick="deleteRow('${rowId}')" class="text-red-400 hover:text-red-600 transition"><i class="fa-solid fa-trash-can"></i></button></td>`;
            ['table-body-exp', 'table-body-anim'].forEach(id => {
                const tr = document.createElement('tr'); tr.id = rowId + '-' + id; tr.dataset.group = rowId;
                tr.className = "border-b transition-colors hover:bg-gray-50"; tr.style.borderColor = "var(--border-color)";
                tr.innerHTML = rowHtml; document.getElementById(id).appendChild(tr);
            });
        }
        window.deleteRow = function(id) { document.querySelectorAll(`tr[data-group="${id}"]`).forEach(el => el.remove()); }
        window.clearData = function() {
            if(confirm('确定要清空所有数据吗？')) {
                document.getElementById('table-body-exp').innerHTML = '';
                document.getElementById('table-body-anim').innerHTML = '';
                recordCount = 0;
            }
        }
        function resize() { canvas.width = canvas.parentElement.clientWidth; canvas.height = canvas.parentElement.clientHeight; state.pxPerCm = canvas.width / 85; draw(); }

        function drawClippedLine(x1, y1, slope, dashRanges, color, arrowAt) {
            const w = canvas.width, h = canvas.height;
            let points = [];
            let yAt0 = y1 + slope * (0 - x1);
            if(yAt0 >= 0 && yAt0 <= h) points.push({x:0, y:yAt0});
            let yAtW = y1 + slope * (w - x1);
            if(yAtW >= 0 && yAtW <= h) points.push({x:w, y:yAtW});
            if(slope !== 0) {
                let xAt0 = x1 + (0 - y1) / slope;
                if(xAt0 > 0 && xAt0 < w) points.push({x:xAt0, y:0});
            }
            if(slope !== 0) {
                let xAtH = x1 + (h - y1) / slope;
                if(xAtH > 0 && xAtH < w) points.push({x:xAtH, y:h});
            }
            points.sort((a,b) => a.x - b.x);
            if(points.length < 2) return;
            const start = points[0];
            const end = points[points.length-1];
            ctx.beginPath();
            dashRanges.forEach(range => {
                const drawStart = Math.max(start.x, range.minX);
                const drawEnd = Math.min(end.x, range.maxX);
                if(drawEnd > drawStart) {
                    const yS = y1 + slope * (drawStart - x1);
                    const yE = y1 + slope * (drawEnd - x1);
                    ctx.beginPath();
                    if(range.dashed) {
                        ctx.setLineDash([8, 8]); ctx.lineWidth = 3; ctx.strokeStyle = themeConfig[currentTheme].rayVirtual;
                    } else {
                        ctx.setLineDash([]); ctx.lineWidth = 2.5; ctx.strokeStyle = color;
                    }
                    ctx.moveTo(drawStart, yS); ctx.lineTo(drawEnd, yE); ctx.stroke();
                    ctx.setLineDash([]);
                    if(arrowAt >= drawStart && arrowAt <= drawEnd && !range.dashed) {
                        const yA = y1 + slope * (arrowAt - x1);
                        drawArrowHead(arrowAt, yA, Math.atan(slope), color);
                    }
                }
            });
        }

        function draw() {
            const { width, height } = canvas;
            const cx = width / 2, cy = height / 2;
            const { f, u, pxPerCm, candleH } = state;
            const colors = themeConfig[currentTheme];
            ctx.clearRect(0, 0, width, height);
            ctx.beginPath(); ctx.moveTo(0, cy); ctx.lineTo(width, cy); ctx.strokeStyle = '#D6D3D1'; ctx.lineWidth = 2; ctx.stroke();
            const curvature = 250 / f; 
            drawJellyLens(cx, cy, 220, curvature, colors);
            const fPx = f * pxPerCm;
            drawPoint(cx - fPx, cy, "F"); drawPoint(cx + fPx, cy, "F");
            drawPoint(cx - 2*fPx, cy, "2F"); drawPoint(cx + 2*fPx, cy, "2F");
            const objX = cx - u * pxPerCm;
            const objY = cy - candleH;
            drawKawaiiCandle(objX, cy, candleH, 1, false, colors, true);
            ctx.fillStyle = '#EA580C'; ctx.font = 'bold 12px Comic Sans MS'; ctx.textAlign = 'center'; ctx.fillText(`u=${u.toFixed(1)}`, objX, cy + 20);
            if (Math.abs(u - f) < 0.1) {
                drawParallel(cx, cy, objX, objY);
            } else {
                const slopeRed = (cy - objY) / fPx;
                const slopeGreen = (cy - objY) / (cx - objX);
                const ix = cx + (cy - objY) / (slopeRed - slopeGreen);
                const iy = cy + slopeGreen * (ix - cx);
                const isReal = ix > cx;
                ctx.strokeStyle = '#FB7185'; ctx.lineWidth = 2.5; ctx.beginPath(); ctx.moveTo(objX, objY); ctx.lineTo(cx, objY); ctx.stroke();
                drawArrowHead((objX+cx)/2, objY, 0, '#FB7185');
                let redRanges = [];
                if(isReal) {
                    redRanges.push({minX: cx, maxX: 100000, dashed: false});
                } else {
                    redRanges.push({minX: -100000, maxX: cx, dashed: true});
                    redRanges.push({minX: cx, maxX: 100000, dashed: false});
                }
                drawClippedLine(cx, objY, slopeRed, redRanges, '#FB7185', cx + 60);
                let greenRanges = [];
                if(isReal) {
                    greenRanges.push({minX: objX, maxX: 100000, dashed: false});
                } else {
                    greenRanges.push({minX: -100000, maxX: objX, dashed: true});
                    greenRanges.push({minX: objX, maxX: 100000, dashed: false});
                }
                drawClippedLine(cx, cy, slopeGreen, greenRanges, '#4ADE80', cx + 60);
                drawArrowHead((objX+cx)/2, (objY+cy)/2, Math.atan(slopeGreen), '#4ADE80');
                const imgH = Math.abs(iy - cy);
                if(Math.abs(ix) < 10000 && imgH < 5000) {
                     const vVal = (ix - cx) / pxPerCm;
                     const mVal = imgH / candleH;
                     const drawH = Math.min(imgH, height * 0.8);
                     if(ix > -width*2 && ix < width*2) {
                         drawKawaiiCandle(ix, cy, drawH, isReal?0.9:0.4, isReal, colors, true);
                         ctx.fillStyle = isReal ? '#E11D48' : '#D97706'; 
                         ctx.fillText(`v=${Math.abs(vVal).toFixed(1)}`, ix, cy + (isReal?-10:20));
                     }
                }
            }
        }

        function drawParallel(cx, cy, objX, objY) {
            const fPx = state.f * state.pxPerCm; ctx.lineWidth = 2.5; ctx.lineCap = 'round';
            ctx.strokeStyle = '#FB7185'; ctx.beginPath(); ctx.moveTo(objX, objY); ctx.lineTo(cx, objY); ctx.stroke();
            drawArrowHead((objX + cx)/2, objY, 0, '#FB7185');
            const slope = (cy - objY) / fPx;
            drawClippedLine(cx, objY, slope, [{minX: cx, maxX: 100000, dashed: false}], '#FB7185', cx + 60);
            const slopeG = (cy - objY) / (cx - objX);
            drawClippedLine(cx, cy, slopeG, [{minX: objX, maxX: 100000, dashed: false}], '#4ADE80', cx + 60);
            drawArrowHead((objX+cx)/2, (objY+cy)/2, Math.atan(slopeG), '#4ADE80');
        }

        function drawJellyLens(cx, cy, h, curve, colors) {
            ctx.save(); ctx.fillStyle = colors.lensFill; ctx.strokeStyle = colors.lensStroke; ctx.lineWidth = 2;
            ctx.beginPath(); ctx.moveTo(cx, cy - h/2); ctx.quadraticCurveTo(cx + curve, cy, cx, cy + h/2); ctx.quadraticCurveTo(cx - curve, cy, cx, cy - h/2);
            ctx.fill(); ctx.stroke(); ctx.fillStyle = colors.lensStroke; ctx.font = 'bold 12px Comic Sans MS'; ctx.fillText("O", cx - 4, cy + h/2 + 20); ctx.restore();
        }
        function drawPoint(x, y, label) { ctx.beginPath(); ctx.arc(x, y, 4, 0, Math.PI*2); ctx.fillStyle = '#F43F5E'; ctx.fill(); ctx.fillStyle = '#9CA3AF'; ctx.font = '10px Arial'; ctx.textAlign = 'center'; ctx.fillText(label, x, y + 16); }
        function drawArrowHead(x, y, angle, color) { ctx.save(); ctx.translate(x, y); ctx.rotate(angle); ctx.beginPath(); ctx.moveTo(0, 0); ctx.lineTo(-10, 4); ctx.lineTo(-10, -4); ctx.closePath(); ctx.fillStyle = color; ctx.fill(); ctx.restore(); }
        
        function drawKawaiiCandle(x, y, h, alpha, isInverted, colors, animate) {
            ctx.save(); ctx.globalAlpha = alpha; ctx.translate(x, y); if (isInverted) ctx.scale(1, -1);
            const w = h * 0.55, waxH = h * 0.7, flameH = h * 0.3, radius = w * 0.3, topY = -waxH, leftX = -w/2;
            const defineBodyPath = () => {
                ctx.beginPath(); ctx.moveTo(leftX, topY + radius); ctx.quadraticCurveTo(leftX, topY, leftX + radius, topY);
                ctx.lineTo(w/2 - radius, topY); ctx.quadraticCurveTo(w/2, topY, w/2, topY + radius);
                ctx.lineTo(w/2, -radius); ctx.quadraticCurveTo(w/2, 0, w/2 - radius, 0);
                ctx.lineTo(leftX + radius, 0); ctx.quadraticCurveTo(leftX, 0, leftX, -radius); ctx.closePath();
            };
            defineBodyPath(); ctx.save(); ctx.clip(); ctx.fillStyle = '#FFFFFF'; ctx.fill();
            ctx.lineWidth = w * 0.35; ctx.strokeStyle = colors.candleStripe; ctx.lineCap = 'round';
            for(let i = -2; i < 8; i++) { const startY = topY + i * ((waxH+w)/4); ctx.beginPath(); ctx.moveTo(leftX-w, startY); ctx.quadraticCurveTo(leftX+w/1.5, startY+w*1.5, leftX+w*2.5, startY+w*2.5); ctx.stroke(); }
            ctx.beginPath(); ctx.moveTo(leftX + w*0.15, topY + radius); ctx.lineTo(leftX + w*0.15, -radius); ctx.lineWidth = w * 0.08; ctx.strokeStyle = 'rgba(255, 255, 255, 0.6)'; ctx.lineCap = 'round'; ctx.stroke(); ctx.restore(); 
            defineBodyPath(); ctx.strokeStyle = colors.candleStroke; ctx.lineWidth = 2; ctx.stroke();
            ctx.beginPath(); ctx.moveTo(0, topY); ctx.quadraticCurveTo(0, topY - flameH*0.2, w*0.05, topY - flameH*0.2); ctx.strokeStyle = '#78716C'; ctx.lineWidth = 3; ctx.lineCap = 'round'; ctx.stroke();
            const flameBaseY = topY - flameH * 0.15, t = Date.now();
            const sway = animate ? (Math.sin(t/200)*w*0.1 + Math.sin(t/50)*w*0.05) : 0, breathe = animate ? Math.sin(t/150)*flameH*0.05 : 0;
            const grad = ctx.createRadialGradient(0, flameBaseY - flameH*0.5, 0, 0, flameBaseY - flameH*0.5, flameH);
            grad.addColorStop(0, '#FEF08A'); grad.addColorStop(0.4, '#FACC15'); grad.addColorStop(1, '#EA580C');
            ctx.beginPath(); ctx.moveTo(0, flameBaseY); 
            ctx.bezierCurveTo(-w*0.5 + sway, flameBaseY - flameH*0.4, -w*0.1 + sway*1.5, flameBaseY - flameH*1.1 - breathe, sway*2, flameBaseY - flameH*1.1 - breathe);
            ctx.bezierCurveTo(w*0.1 + sway*1.5, flameBaseY - flameH*1.1 - breathe, w*0.5 + sway, flameBaseY - flameH*0.4, 0, flameBaseY);
            ctx.fillStyle = grad; ctx.fill(); ctx.lineWidth = 1; ctx.strokeStyle = 'rgba(255,255,255,0.5)'; ctx.stroke(); ctx.restore();
        }

        function startDrag(e) { stopAnimation(); const rect = canvas.getBoundingClientRect(), x = (e.touches?e.touches[0].clientX:e.clientX)-rect.left, cx = canvas.width/2; if(Math.abs(x - (cx - state.u * state.pxPerCm)) < 60) state.isDragging = true; }
        function onDrag(e) { if(!state.isDragging) return; e.preventDefault(); const rect = canvas.getBoundingClientRect(), x = (e.touches?e.touches[0].clientX:e.clientX)-rect.left, cx = canvas.width/2; state.u = Math.max(2, Math.min(60, (cx - x) / state.pxPerCm)); updateAllUI(); draw(); }
        function endDrag() { state.isDragging = false; }
