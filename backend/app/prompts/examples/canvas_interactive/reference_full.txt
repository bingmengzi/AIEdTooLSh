```html
            <i class="fa-solid fa-rotate-right mr-1"></i> 重置
        </button>
    </header>

    <script>
        // === 颜色常量 ===
        const COLORS = {
            primary: '#4285F4',   // 蓝 - 主曲线
            accent: '#EA4335',    // 红 - 焦点
            secondary: '#64748b', // 灰 - 准线
            success: '#34A853',   // 绿 - 辅助
            warning: '#FBBC05',   // 黄 - 动画点
            grid: '#f0f0f0',
            axis: '#cbd5e1'
        };

        // === 状态管理 ===
        const canvas = document.getElementById('mainCanvas');
        const ctx = canvas.getContext('2d');
        const container = document.getElementById('canvasContainer');

        let state = {
            p: 2.0,                    // 焦距，必须非零
            isDraggingFocus: false,
            isDraggingDirectrix: false,
            dragOffset: 0,
            animPointT: 0              // 动画参数
        };

        // DOM 元素
        const pSlider = document.getElementById('pSlider');
        const pValueDisplay = document.getElementById('pValueDisplay');
        const equationDisplay = document.getElementById('equationDisplay');
        const focusCoordDisplay = document.getElementById('focusCoord');
        const directrixCoordDisplay = document.getElementById('directrixCoord');

        // === 初始化 ===
        function init() {
            resizeCanvas();
            window.addEventListener('resize', resizeCanvas);

            pSlider.addEventListener('input', (e) => {
                state.p = parseFloat(e.target.value);
                updateUI();
                draw();
            });

            // 鼠标事件
            canvas.addEventListener('mousedown', handleInputStart);
            canvas.addEventListener('mousemove', handleInputMove);
            window.addEventListener('mouseup', handleInputEnd);

            // 触摸事件
            canvas.addEventListener('touchstart', (e) => { e.preventDefault(); handleInputStart(e.touches[0]); }, {passive: false});
            canvas.addEventListener('touchmove', (e) => { e.preventDefault(); handleInputMove(e.touches[0]); }, {passive: false});
            window.addEventListener('touchend', handleInputEnd);

            requestAnimationFrame(animate);
            updateUI();
            draw();
        }

        // === DPI 适配 ===
        function resizeCanvas() {
            const rect = container.getBoundingClientRect();
            const dpr = window.devicePixelRatio || 1;
            canvas.width = rect.width * dpr;
            canvas.height = rect.height * dpr;
            canvas.style.width = `${rect.width}px`;
            canvas.style.height = `${rect.height}px`;
            ctx.scale(dpr, dpr);
            draw();
        }

        // === 坐标转换 ===
        function getCoords() {
            const rect = canvas.getBoundingClientRect();
            return { w: rect.width, h: rect.height, cx: rect.width / 2, cy: rect.height / 2 };
        }

        function toScreen(x, y, c) {
            const scale = Math.min(c.w, c.h) / 12;
            return { x: c.cx + x * scale, y: c.cy - y * scale };
        }

        function toLogic(sx, sy, c) {
            const scale = Math.min(c.w, c.h) / 12;
            return { x: (sx - c.cx) / scale, y: -(sy - c.cy) / scale };
        }

        // === 主绘图函数 ===
        function draw() {
            const c = getCoords();
            ctx.clearRect(0, 0, c.w, c.h);

            drawGrid(c);
            drawAxes(c);
            drawDirectrix(c);
            drawParabola(c);
            drawFocus(c);
            drawAnimatedPoint(c);
        }

        function drawGrid(c) {
            ctx.strokeStyle = COLORS.grid;
            ctx.lineWidth = 1;
            const step = Math.min(c.w, c.h) / 12;
            ctx.beginPath();
            for (let i = 0; i <= c.w; i += step) { ctx.moveTo(i, 0); ctx.lineTo(i, c.h); }
            for (let i = 0; i <= c.h; i += step) { ctx.moveTo(0, i); ctx.lineTo(c.w, i); }
            ctx.stroke();
        }

        function drawAxes(c) {
            ctx.strokeStyle = COLORS.axis;
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.moveTo(0, c.cy); ctx.lineTo(c.w, c.cy);
            ctx.moveTo(c.cx, 0); ctx.lineTo(c.cx, c.h);
            ctx.stroke();
            ctx.fillStyle = COLORS.secondary;
            ctx.font = '12px Arial';
            ctx.fillText('O', c.cx + 5, c.cy + 15);
        }

        function drawDirectrix(c) {
            ctx.save();
            ctx.strokeStyle = COLORS.secondary;
            ctx.setLineDash([8, 8]);
            ctx.lineWidth = 2;
            const pos = toScreen(-state.p, 0, c).x;
            ctx.beginPath();
            ctx.moveTo(pos, 0);
            ctx.lineTo(pos, c.h);
            ctx.stroke();
            ctx.setLineDash([]);
            ctx.fillText(`x = ${(-state.p).toFixed(1)}`, pos + 5, 20);
            ctx.restore();
        }

        function drawParabola(c) {
            ctx.save();
            ctx.strokeStyle = COLORS.primary;
            ctx.lineWidth = 3;
            ctx.lineCap = 'round';
            ctx.beginPath();
            let started = false;
            for (let y = -10; y <= 10; y += 0.1) {
                const x = (y * y) / (4 * state.p);
                const pos = toScreen(x, y, c);
                if (!started) { ctx.moveTo(pos.x, pos.y); started = true; }
                else { ctx.lineTo(pos.x, pos.y); }
            }
            ctx.stroke();
            ctx.restore();
        }

        function drawFocus(c) {
            const pos = toScreen(state.p, 0, c);
            // 光晕
            const gradient = ctx.createRadialGradient(pos.x, pos.y, 2, pos.x, pos.y, 15);
            gradient.addColorStop(0, 'rgba(234, 67, 53, 0.4)');
            gradient.addColorStop(1, 'rgba(234, 67, 53, 0)');
            ctx.fillStyle = gradient;
            ctx.beginPath();
            ctx.arc(pos.x, pos.y, 15, 0, Math.PI * 2);
            ctx.fill();
            // 实心点
            ctx.fillStyle = COLORS.accent;
            ctx.beginPath();
            ctx.arc(pos.x, pos.y, 6, 0, Math.PI * 2);
            ctx.fill();
            // 标签
            ctx.font = 'bold 14px Arial';
            ctx.fillText('F (焦点)', pos.x + 10, pos.y - 10);
        }

        function drawAnimatedPoint(c) {
            const t = state.animPointT;
            const px = state.p * t * t;
            const py = 2 * state.p * t;
            const dx = -state.p, dy = py;
            const fx = state.p, fy = 0;

            const pPos = toScreen(px, py, c);
            const dPos = toScreen(dx, dy, c);
            const fPos = toScreen(fx, fy, c);

            // 连线
            ctx.save();
            ctx.setLineDash([4, 4]);
            ctx.lineWidth = 1.5;
            ctx.strokeStyle = COLORS.accent;
            ctx.beginPath(); ctx.moveTo(pPos.x, pPos.y); ctx.lineTo(fPos.x, fPos.y); ctx.stroke();
            ctx.strokeStyle = COLORS.success;
            ctx.beginPath(); ctx.moveTo(pPos.x, pPos.y); ctx.lineTo(dPos.x, dPos.y); ctx.stroke();
            ctx.restore();

            // 动画点 (黄色+白边)
            ctx.fillStyle = COLORS.warning;
            ctx.beginPath();
            ctx.arc(pPos.x, pPos.y, 5, 0, Math.PI * 2);
            ctx.fill();
            ctx.strokeStyle = '#fff';
            ctx.lineWidth = 2;
            ctx.stroke();

            // 距离标签
            ctx.fillStyle = COLORS.primary;
            ctx.font = '12px Arial';
            ctx.textAlign = 'center';
            ctx.fillText('|PF|', (pPos.x + fPos.x) / 2 - 10, (pPos.y + fPos.y) / 2 - 5);
            ctx.fillText('|PL|', (pPos.x + dPos.x) / 2 + 10, (pPos.y + dPos.y) / 2 + 5);
        }

        // === 动画循环 ===
        function animate() {
            state.animPointT += 0.02;
            if (state.animPointT > 3.5) state.animPointT = -3.5;
            draw();
            requestAnimationFrame(animate);
        }

        // === 交互处理 ===
        function handleInputStart(e) {
            const c = getCoords();
            const rect = canvas.getBoundingClientRect();
            const logicPos = toLogic(e.clientX - rect.left, e.clientY - rect.top, c);

            const distToF = Math.sqrt((logicPos.x - state.p)**2 + logicPos.y**2);
            if (distToF < 1.0) { state.isDraggingFocus = true; return; }

            const distToD = Math.abs(logicPos.x - (-state.p));
            if (distToD < 1.0) {
                state.isDraggingDirectrix = true;
                state.dragOffset = logicPos.x - (-state.p);
            }
        }

        function handleInputMove(e) {
            if (!state.isDraggingFocus && !state.isDraggingDirectrix) return;
            const c = getCoords();
            const rect = canvas.getBoundingClientRect();
            const logicPos = toLogic(e.clientX - rect.left, e.clientY - rect.top, c);

            if (state.isDraggingFocus) {
                state.p = Math.max(0.5, Math.min(5, logicPos.x));
            } else if (state.isDraggingDirectrix) {
                state.p = Math.max(0.5, Math.min(5, -(logicPos.x - state.dragOffset)));
            }
            pSlider.value = state.p;
            updateUI();
            draw();
        }

        function handleInputEnd() {
            state.isDraggingFocus = false;
            state.isDraggingDirectrix = false;
        }

        // === UI 更新 ===
        function updateUI() {
            const p = state.p.toFixed(1);
            pValueDisplay.textContent = p;
            equationDisplay.textContent = `y² = ${(4 * state.p).toFixed(1)}x`;
            focusCoordDisplay.textContent = `(${p}, 0)`;
            directrixCoordDisplay.textContent = `x = ${(-state.p).toFixed(1)}`;
        }

        function resetSimulation() {
            state.p = 2.0;
            state.animPointT = 0;
            pSlider.value = 2;
            updateUI();
            draw();
        }

        init();
    </script>
</body>
</html>
```

## 你生成的代码必须包含以上参考案例的所有关键特征：

1. **完整HTML结构** - DOCTYPE → html → head(meta+CDN) → body
2. **精美布局** - Tailwind三段式：header + main(canvas左 + panel右)
3. **自定义CSS** - body背景、滑块样式、canvas样式
4. **COLORS常量** - 脚本开头定义颜色常量
5. **状态管理** - state对象集中管理参数和交互状态
6. **DPI适配** - resizeCanvas()处理高清屏
7. **坐标转换** - toScreen/toLogic函数
8. **分层绘图** - draw()按顺序调用drawGrid/drawAxes/drawCurve/drawPoints
9. **曲线绘制** - 参数方程for循环遍历
10. **关键点光晕** - createRadialGradient效果
11. **动画循环** - requestAnimationFrame + animPointT递增
12. **完整交互** - 鼠标+触摸事件，拖拽与滑块双向同步
13. **知识点卡片** - 渐变背景+彩色圆点列表
