    function nextStep() {
        if (currentStep >= stepsData.length) {

            btn.className = "bg-slate-400 text-white px-6 py-2.5 rounded-xl font-bold cursor-not-allowed flex items-center gap-2";
            return;
        }

        const step = stepsData[currentStep];
        const container = document.getElementById('steps-container');

        // 构建卡片HTML
        let colorClass = `text-${step.color}`;
        if(step.color === 'eduBlue') colorClass = 'text-[#4285F4]';
        if(step.color === 'eduGreen') colorClass = 'text-[#34A853]';
        if(step.color === 'eduRed') colorClass = 'text-[#EA4335]';
        if(step.color === 'eduYellow') colorClass = 'text-[#FBBC05]';

        const cardHTML = `
            <div class="bg-white rounded-xl shadow-md p-5 border-t-4 border-${step.color.replace('edu', '').toLowerCase()}-500 animate-fade-in-up">
                <h3 class="font-bold text-lg mb-3 flex items-center gap-2 ${colorClass}">

            </div>

        // 插入并滚动
        container.insertAdjacentHTML('beforeend', cardHTML);
        setTimeout(() => {
            container.scrollTop = container.scrollHeight;

        // 触发 MathJax 渲染公式（必须检查函数是否存在，因为async加载可能尚未完成）

            MathJax.typesetPromise([container]);
        }

        // 触发对应的 3D 交互钩子
        if (step.onRender) {
            step.onRender();
        }

        currentStep++;

        if (currentStep >= stepsData.length) {
            const btn = document.getElementById('next-btn');
            btn.innerHTML = '<i class="fas fa-check"></i> 推演完成';
            btn.classList.remove('bg-eduBlue', 'hover:bg-blue-600', 'hover:-translate-y-0.5');

    }


    // ==========================================
    // 2. Three.js 3D 建模与交互逻辑
    // ==========================================
    const container = document.getElementById('canvas-container');
    let scene, camera, renderer, controls;
    const labels = [];
    let axesHelper;
    let planePCE, planeBPC; // 缓存面对象以便高亮

    initThree();
    animate();

    function initThree() {
        // 场景与相机
        scene = new THREE.Scene();
        scene.background = new THREE.Color(0xffffff);
        
        // ⚠️ 容器尺寸检查（防止高度塌陷为0导致空白）
        let cWidth = container.clientWidth || container.offsetWidth || 800;
        let cHeight = container.clientHeight || container.offsetHeight || 600;
        if (cHeight <= 0) {
            container.style.height = '100%';
            cHeight = container.clientHeight;
            if (cHeight <= 0) {
                container.style.height = '60vh';
                cHeight = container.clientHeight || 400;
            }
        }
        
        // 数学习惯：Z轴朝上。所以改变相机up向量
        camera = new THREE.PerspectiveCamera(45, cWidth / cHeight, 0.1, 100);
        camera.up.set(0, 0, 1); 
        resetCamera();

        // 渲染器
        renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        renderer.setSize(cWidth, cHeight);
        renderer.setPixelRatio(window.devicePixelRatio);
        container.appendChild(renderer.domElement);

        // 控制器
        controls = new THREE.OrbitControls(camera, renderer.domElement);
        controls.enableDamping = true;
        controls.dampingFactor = 0.05;
        controls.target.set(0, 0, 0);

        // 材质定义
        const matLineDark = new THREE.LineBasicMaterial({ color: 0x334155, linewidth: 2 });
        const matLineBlue = new THREE.LineBasicMaterial({ color: 0x4285F4, linewidth: 2 });
        const matLineDashed = new THREE.LineDashedMaterial({ color: 0x94A3B8, dashSize: 0.2, gapSize: 0.1 });

        // 数据点坐标设定 (r=2)
        const r = 2;
        const pts = {
            O: new THREE.Vector3(0, 0, 0),
            C: new THREE.Vector3(-r, 0, 0),
            A: new THREE.Vector3(r/2, -r*Math.sqrt(3)/2, 0), // 1, -1.732, 0
            B: new THREE.Vector3(r/2, r*Math.sqrt(3)/2, 0),  // 1, 1.732, 0
            E: new THREE.Vector3(-r/2, r*Math.sqrt(3)/2, 0), // -1, 1.732, 0
            D: new THREE.Vector3(0, 0, r*Math.sqrt(3)),      // 0, 0, 3.464
            P: new THREE.Vector3(0, 0, r*Math.sqrt(2)/2)     // 0, 0, 1.414
        };

        // 1. 画底面圆 (只画边缘线，淡淡的)
        const circleGeo = new THREE.BufferGeometry();
        const circlePts = [];
        for (let i = 0; i <= 64; i++) {
            const theta = (i / 64) * Math.PI * 2;
            circlePts.push(r * Math.cos(theta), r * Math.sin(theta), 0);
        }
        circleGeo.setAttribute('position', new THREE.Float32BufferAttribute(circlePts, 3));
        const circleLine = new THREE.Line(circleGeo, new THREE.LineBasicMaterial({color: 0xCBD5E1}));
        scene.add(circleLine);

        // 2. 辅助构建线段的函数
        function addLine(v1, v2, material) {
            const geo = new THREE.BufferGeometry().setFromPoints([v1, v2]);
            const line = new THREE.Line(geo, material);
            if(material.isLineDashedMaterial) line.computeLineDistances();
            scene.add(line);
        }

        // 画外轮廓圆锥虚线 (示例几条母线)
        addLine(pts.D, new THREE.Vector3(r,0,0), matLineDashed);
        addLine(pts.D, new THREE.Vector3(0,r,0), matLineDashed);
        addLine(pts.D, new THREE.Vector3(0,-r,0), matLineDashed);
        addLine(pts.D, pts.C, matLineDashed);

        // 画底面正三角形 ABC
        addLine(pts.A, pts.B, matLineDark);
        addLine(pts.B, pts.C, matLineDark);
        addLine(pts.C, pts.A, matLineDashed); // 视线遮挡用虚线模拟

        // 画直径 AE 和 高 DO
        addLine(pts.A, pts.E, matLineDashed);
        addLine(pts.D, pts.O, matLineDashed);

        // 画正三棱锥核心线 P-ABC
        addLine(pts.P, pts.A, matLineBlue);
        addLine(pts.P, pts.B, matLineBlue);
        addLine(pts.P, pts.C, matLineBlue);

        // 连接 CE
        addLine(pts.C, pts.E, matLineDark);

        // 创建高亮面 BPC (半透明黄)
        const geoBPC = new THREE.BufferGeometry().setFromPoints([pts.B, pts.P, pts.C]);
        planeBPC = new THREE.Mesh(geoBPC, new THREE.MeshBasicMaterial({ color: 0xFBBC05, side: THREE.DoubleSide, transparent: true, opacity: 0, depthWrite: false }));
        scene.add(planeBPC);

        // 创建高亮面 PCE (半透明绿)
        const geoPCE = new THREE.BufferGeometry().setFromPoints([pts.P, pts.C, pts.E]);
        planePCE = new THREE.Mesh(geoPCE, new THREE.MeshBasicMaterial({ color: 0x34A853, side: THREE.DoubleSide, transparent: true, opacity: 0, depthWrite: false }));
        scene.add(planePCE);

        // 添加HTML标签
        addLabel('A', pts.A);
        addLabel('B', pts.B);
        addLabel('C', pts.C);
        addLabel('D', pts.D);
        addLabel('E', pts.E);
        addLabel('O', pts.O);
        addLabel('P', pts.P);

        // 坐标轴辅助 (默认隐藏，Step 4 显示)
        axesHelper = new THREE.AxesHelper(3.5);
        axesHelper.visible = false;
        scene.add(axesHelper);

        // 监听窗口大小变化
        window.addEventListener('resize', onWindowResize, false);
    }

    function addLabel(text, pos) {
        const div = document.createElement('div');
        div.className = 'scene-label';
        div.innerHTML = text;
        // 给特定点上色强调
        if(text==='P') div.style.color = '#EA4335';
        if(text==='O') div.style.color = '#94A3B8';
        container.appendChild(div);
        labels.push({ elem: div, pos: pos });
    }

    function updateLabels() {
        labels.forEach(label => {
            const vector = label.pos.clone();
            vector.project(camera);
            // 转换为屏幕坐标
            const x = (vector.x * 0.5 + 0.5) * container.clientWidth;
            const y = (vector.y * -0.5 + 0.5) * container.clientHeight;
            
            if (vector.z > 1) { // 剔除在相机背后的点
                label.elem.style.opacity = '0';
            } else {
                label.elem.style.opacity = '1';
                label.elem.style.left = `${x}px`;
                label.elem.style.top = `${y}px`;
            }
        });
    }

    function animate() {
        requestAnimationFrame(animate);
        controls.update();
        renderer.render(scene, camera);
        updateLabels();
    }

    function onWindowResize() {
        if(!container) return;
        // ⚠️ 同样需要容器尺寸检查
        let w = container.clientWidth || container.offsetWidth;
        let h = container.clientHeight || container.offsetHeight;
        if (h <= 0) {
            container.style.height = '100%';
            h = container.clientHeight;
            if (h <= 0) {
                container.style.height = '60vh';
                h = container.clientHeight || 400;
            }
        }
        camera.aspect = w / h;
        camera.updateProjectionMatrix();
        renderer.setSize(w, h);
    }

    // --- 视角与交互控制函数 ---
    function resetCamera() {
        // 侧上方俯视，能清晰看到P点和底面
        camera.position.set(4, -5, 4);
        if(controls) controls.target.set(0, 0, 1);
    }

    function topViewCamera() {
        // 正上方俯视底面，体现建系原理
        camera.position.set(0, 0.1, 7); // y给个微小偏移防止up轴奇异
        if(controls) controls.target.set(0, 0, 0);
    }

    // 步骤相关的模型特效
    function showAxes() {
        axesHelper.visible = true;
        topViewCamera(); // 自动切换到俯视帮助理解坐标系
    }

    function highlightPCE() {
        planePCE.material.opacity = 0.4;
    }

    function highlightBothPlanes() {
        resetCamera();
        planeBPC.material.opacity = 0.4;
        planePCE.material.opacity = 0.4;
    }
